# 循环神经网络（上）SRN / LSTM 与梯度截断

本节用一个**数字记忆任务**比较 SRN（最简单循环网络）和 LSTM 的长程依赖能力，并演示梯度截断。

**任务**：输入长度 $L$ 的数字序列（每位 ∈ {0..9}），预测**前两位之和**（0..18 共 19 类）。模型必须把开头信息一直保留到序列末尾。

## 1. 数据

```python
import math
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset
import matplotlib.pyplot as plt

def make_digit_sum(n, L, seed=0):
    g = torch.Generator().manual_seed(seed)
    X = torch.randint(0, 10, (n, L), generator=g)
    y = (X[:, 0] + X[:, 1]).long()
    return X, y

X, y = make_digit_sum(5, L=10)
for xi, yi in zip(X, y):
    print(f'seq {xi.tolist()}  ->  label {yi.item()} (= {xi[0].item()}+{xi[1].item()})')

import sys, os
sys.path.insert(0, os.path.abspath(os.path.join(os.getcwd(), os.pardir)))

```

## 2. SRN 从头实现

递推：$h_t = \tanh(W_x x_t + W_h h_{t-1} + b)$。

```python
class MySRN(nn.Module):
    def __init__(self, vocab=10, embed=32, hidden=64, n_class=19):
        super().__init__()
        self.emb = nn.Embedding(vocab, embed)
        self.Wx = nn.Parameter(torch.randn(embed,  hidden) / math.sqrt(embed))
        self.Wh = nn.Parameter(torch.eye(hidden))                    # identity init：给 SRN 加 buff，保护长程信息；
        # 实战中更常见的是 randn / sqrt(hidden)，效果会显著差于 LSTM
        self.b  = nn.Parameter(torch.zeros(hidden))
        self.fc = nn.Linear(hidden, n_class)

    def forward(self, x):                  # x: [B, L]
        e = self.emb(x)                    # [B, L, embed]
        B, L, _ = e.shape
        h = torch.zeros(B, self.Wh.size(0))
        for t in range(L):
            h = torch.tanh(e[:, t] @ self.Wx + h @ self.Wh + self.b)
        return self.fc(h)

print('output shape:', MySRN()(torch.zeros(2, 10, dtype=torch.long)).shape)
```

### 与 `nn.RNN` 对齐

PyTorch 的 `nn.RNN` 权重布局：`weight_ih_l0` 形状 `[hidden, input]`、`weight_hh_l0` 形状 `[hidden, hidden]`——与手写的 `Wx (input, hidden)`、`Wh (hidden, hidden)` 互为转置。

```python
torch.manual_seed(0)
embed_dim, hidden_dim = 16, 32
emb = nn.Embedding(10, embed_dim)

Wx = torch.randn(embed_dim, hidden_dim) / math.sqrt(embed_dim)
Wh = torch.eye(hidden_dim)
b  = torch.zeros(hidden_dim)

x = torch.randint(0, 10, (2, 5))
e = emb(x)
h = torch.zeros(2, hidden_dim)
for t in range(5):
    h = torch.tanh(e[:, t] @ Wx + h @ Wh + b)
h_manual = h

rnn = nn.RNN(embed_dim, hidden_dim, batch_first=True, nonlinearity='tanh')
with torch.no_grad():
    rnn.weight_ih_l0.copy_(Wx.T)
    rnn.weight_hh_l0.copy_(Wh.T)
    # 注意：nn.RNN 的有效 bias = bias_ih + bias_hh；这里手写 b=zeros，所以两边都置零
    # 如果未来把手写 b 改成非零值，需要相应分配（例如 bias_ih = b, bias_hh = zeros）
    rnn.bias_ih_l0.zero_()
    rnn.bias_hh_l0.zero_()
_, h_final = rnn(e)
print('max abs diff:', (h_manual - h_final.squeeze(0)).abs().max().item())
```

## 3. LSTM

用 `nn.LSTM` 直接调即可。接口与 `nn.RNN` 类似，返回 `(out, (h, c))`。

```python
class MyLSTMModel(nn.Module):
    def __init__(self, vocab=10, embed=32, hidden=64, n_class=19):
        super().__init__()
        self.emb = nn.Embedding(vocab, embed)
        self.lstm = nn.LSTM(embed, hidden, batch_first=True)
        self.fc = nn.Linear(hidden, n_class)

    def forward(self, x):
        e = self.emb(x)
        _, (h, _) = self.lstm(e)
        # h: [num_layers, B, hidden]，取最后一层末步隐状态（1 层时等价于 h.squeeze(0)）
        return self.fc(h[-1])
```

## 4. 训练对比：SRN vs LSTM × 不同序列长度

在 $L \in \{10, 20\}$ 上各训 20 个 epoch，统一用 Adam + grad clip。

```python
from nndl import accuracy

def train(model_factory, L, epochs=20, lr=3e-3, n_train=3000, clip=1.0):
    torch.manual_seed(0)
    model = model_factory()
    opt = optim.Adam(model.parameters(), lr=lr)
    X_tr, y_tr = make_digit_sum(n_train, L, seed=0)
    X_dv, y_dv = make_digit_sum(500,     L, seed=1)
    train_loader = DataLoader(TensorDataset(X_tr, y_tr), batch_size=128, shuffle=True)
    dev_loader   = DataLoader(TensorDataset(X_dv, y_dv), batch_size=256)

    runner = RunnerV3(model, opt, nn.CrossEntropyLoss(),
                      metric_fn=accuracy, higher_is_better=True)
    runner.fit(train_loader, dev_loader, num_epochs=epochs, log_every=None,
               grad_clip_norm=clip)
    return runner.history['dev_metric']

results = {}
for L in [10, 20]:
    print(f'--- L = {L} ---')
    results[('SRN',  L)] = train(MySRN,        L)
    print(f'  SRN  best={max(results[("SRN",  L)]):.3f}  final={results[("SRN",  L)][-1]:.3f}')
    results[('LSTM', L)] = train(MyLSTMModel,  L)
    print(f'  LSTM best={max(results[("LSTM", L)]):.3f}  final={results[("LSTM", L)][-1]:.3f}')
```

## 4b. 复现书中实验：不同长度下的 SRN / LSTM 损失曲线

上面的快速对比用的是简化设置（2 个长度、少量轮次）。这一节按**书中正文的口径**跑完整实验，复现书里“不同长度的 SRN/LSTM 损失曲线”那组图：6 个长度（10、15、20、25、30、35）、每个长度训练 500 轮、训练集 300 条（穷举前两位数字组合 + k 增强），用与书一致的手写 SRN/LSTM 算子。

> ⚠️ 完整跑（`QUICK=False`）在 CPU 上约需 30–45 分钟。建议先用 `QUICK=True`（2 个长度 / 50 轮，约 1 分钟）跑通流程，确认无误后再设 `QUICK=False` 复现全部曲线。

> 此外，正文图中的损失/准确率曲线为 **5 个随机种子独立训练后取平均**，以抑制单次训练的随机波动；本笔记本为简洁起见只跑单个种子，因此单次结果会比书图更跳、个别长度的准确率可能起伏。

```python
import os, random
import numpy as np
from torch.utils.data import Dataset
from nndl import RunnerV3, accuracy

TAB, NL = chr(9), chr(10)

# 数据：穷举前两位 100 种组合，其余位填 0 并随机替换；k=3 训练 / k=5 验证测试
def generate_data(length, k, save_path):
    base = [([n1, n2] + [0] * (length - 2), n1 + n2)
            for n1 in range(10) for n2 in range(10)]
    ex = []
    for seq0, label in base:
        for _ in range(k):
            seq = seq0.copy()
            seq[np.random.randint(2, length)] = np.random.randint(0, 10)
            ex.append((seq, label))
    os.makedirs(os.path.dirname(save_path), exist_ok=True)
    with open(save_path, "w", encoding="utf-8") as f:
        for seq, label in ex:
            f.write(" ".join(map(str, seq)) + TAB + str(label) + NL)

def load_data(p):
    def rd(fn):
        out = []
        for line in open(os.path.join(p, fn), encoding="utf-8"):
            a, b = line.strip().split(TAB)
            out.append(([int(i) for i in a.split()], int(b)))
        return out
    return rd("train.txt"), rd("dev.txt"), rd("test.txt")

class DigitSumDataset(Dataset):
    def __init__(self, d): self.data = d
    def __getitem__(self, i):
        s, l = self.data[i]
        return torch.tensor(s), torch.tensor(l)
    def __len__(self): return len(self.data)

# 手写算子（与书正文一致）
class Embedding(nn.Module):
    def __init__(self, n, d):
        super().__init__(); self.W = nn.Parameter(torch.randn(n, d))
    def forward(self, x): return self.W[x]

class SRN(nn.Module):
    def __init__(self, input_size, hidden_size):
        super().__init__(); self.hidden_size = hidden_size
        # 小尺度初始化（×0.1）：否则标准正态会让 tanh 一开始就饱和、SRN 难训练
        self.W = nn.Parameter(0.1 * torch.randn(input_size, hidden_size))
        self.U = nn.Parameter(0.1 * torch.randn(hidden_size, hidden_size))
        self.b = nn.Parameter(torch.zeros(1, hidden_size))
    def init_state(self, b):
        return next(self.parameters()).new_zeros(b, self.hidden_size)
    def forward(self, inp, hs=None):
        b, L, _ = inp.shape
        hs = self.init_state(b) if hs is None else hs
        for t in range(L):
            hs = torch.tanh(inp[:, t, :] @ self.W + hs @ self.U + self.b)
        return hs

class LSTM(nn.Module):
    def __init__(self, input_size, hidden_size):
        super().__init__(); self.hidden_size = hidden_size
        p = lambda *s: nn.Parameter(torch.randn(*s))
        self.W_i, self.W_f, self.W_o, self.W_c = p(input_size, hidden_size), p(input_size, hidden_size), p(input_size, hidden_size), p(input_size, hidden_size)
        self.U_i, self.U_f, self.U_o, self.U_c = p(hidden_size, hidden_size), p(hidden_size, hidden_size), p(hidden_size, hidden_size), p(hidden_size, hidden_size)
        self.b_i, self.b_f, self.b_o, self.b_c = p(1, hidden_size), p(1, hidden_size), p(1, hidden_size), p(1, hidden_size)
    def init_state(self, b):
        r = next(self.parameters())
        return r.new_zeros(b, self.hidden_size), r.new_zeros(b, self.hidden_size)
    def forward(self, inp, st=None):
        b, L, _ = inp.shape
        hs, cs = self.init_state(b) if st is None else st
        for t in range(L):
            x = inp[:, t, :]
            I = torch.sigmoid(x @ self.W_i + hs @ self.U_i + self.b_i)
            Fg = torch.sigmoid(x @ self.W_f + hs @ self.U_f + self.b_f)
            O = torch.sigmoid(x @ self.W_o + hs @ self.U_o + self.b_o)
            C = torch.tanh(x @ self.W_c + hs @ self.U_c + self.b_c)
            cs = Fg * cs + I * C
            hs = O * torch.tanh(cs)
        return hs

class Model_RNN4SeqClass(nn.Module):
    def __init__(self, rnn, num_digits, input_size, hidden_size, num_classes):
        super().__init__(); self.rnn_model = rnn
        self.embedding = Embedding(num_digits, input_size)
        self.linear = nn.Linear(hidden_size, num_classes)
    def forward(self, x):
        return self.linear(self.rnn_model(self.embedding(x)))
```

### LSTM 门控行为可视化

`nn.LSTM` 是 fused 实现，不直接暴露每个时间步的门控值。下面用训练好的 LSTM 权重**手动逐步复算**输入门 $\boldsymbol I_t$、遗忘门 $\boldsymbol F_t$、输出门 $\boldsymbol O_t$ 和内部状态 $\boldsymbol C_t$（PyTorch 把四组门按 $i,f,g,o$ 顺序打包在 `weight_ih_l0 / weight_hh_l0 / bias_*` 中），再用热力图展示它们随序列的变化：横轴为输入数字，纵轴为隐藏维度，颜色深浅代表数值大小。对应书稿图 `rnn-gate-vis`。

```python
# ===== LSTM 门控行为可视化（非粉色重制；参考 practice-in-paddle 图6.13）=====
import torch
import torch.nn as nn
import torch.optim as optim
import matplotlib.pyplot as plt
from torch.utils.data import DataLoader, TensorDataset

plt.rcParams["font.sans-serif"] = ["SimHei", "Microsoft YaHei", "Noto Sans CJK SC", "DejaVu Sans"]
plt.rcParams["axes.unicode_minus"] = False

CMAP_GATE = "viridis"   # 非粉色感知均匀色图（可改 "cividis"/"Blues"/"Reds"）
CMAP_CELL = "coolwarm"  # 内部状态可正可负，用发散色图（可改 "RdBu_r"/"bwr"）

# 1) 训练一个 LSTM（数字求和：标签=前两位之和，信号集中在序列开头）
torch.manual_seed(0)
gate_model = MyLSTMModel()                          # embed=32, hidden=64, n_class=19
opt = optim.Adam(gate_model.parameters(), lr=3e-3)
loss_fn = nn.CrossEntropyLoss()
Xtr, ytr = make_digit_sum(3000, L=10, seed=0)
loader = DataLoader(TensorDataset(Xtr, ytr), batch_size=128, shuffle=True)
gate_model.train()
for _ in range(20):
    for xb, yb in loader:
        opt.zero_grad()
        loss_fn(gate_model(xb), yb).backward()
        nn.utils.clip_grad_norm_(gate_model.parameters(), 1.0)
        opt.step()

# 2) 用训练好的 nn.LSTM 权重手动复算各门（数值与 fused nn.LSTM 完全一致）
@torch.no_grad()
def lstm_gate_trace(model, x):
    lstm, H = model.lstm, model.lstm.hidden_size
    e = model.emb(x)                                   # [1, L, embed]
    Wih, Whh = lstm.weight_ih_l0, lstm.weight_hh_l0    # [4H, embed], [4H, H]
    bih, bhh = lstm.bias_ih_l0, lstm.bias_hh_l0        # [4H], [4H]
    h = torch.zeros(1, H); c = torch.zeros(1, H)
    Is, Fs, Os, Cs = [], [], [], []
    for t in range(e.size(1)):
        g = e[:, t] @ Wih.T + h @ Whh.T + bih + bhh    # [1, 4H]; 顺序 i, f, g, o
        i  = torch.sigmoid(g[:, 0:H])
        f  = torch.sigmoid(g[:, H:2 * H])
        gg = torch.tanh(g[:, 2 * H:3 * H])
        o  = torch.sigmoid(g[:, 3 * H:4 * H])
        c = f * c + i * gg
        h = o * torch.tanh(c)
        Is.append(i); Fs.append(f); Os.append(o); Cs.append(c)
    mat = lambda lst: torch.cat(lst, dim=0).T.numpy()  # -> [H, seq_len]
    return mat(Is), mat(Fs), mat(Os), mat(Cs), h

# 3) 样例序列上复算 + 自检（手动复算的末隐状态应与 nn.LSTM 一致）
inputs = [6, 7, 0, 0, 1, 0, 0, 0, 0, 0]
x = torch.tensor([inputs])
gate_model.eval()
with torch.no_grad():
    pred = gate_model(x).argmax(-1).item()
    _, (h_ref, _) = gate_model.lstm(gate_model.emb(x))
I, Fg, O, C, h_manual = lstm_gate_trace(gate_model, x)
assert torch.allclose(h_manual, h_ref.squeeze(0), atol=1e-5), "门复算与 nn.LSTM 不一致，请检查门顺序"
print(f"输入 {inputs} -> 预测 {pred}（前两位之和={inputs[0] + inputs[1]}）；门复算自检通过")

# 4) 四联热力图（非粉色）。横轴=输入数字，纵轴=隐藏维度，颜色深浅=数值大小
panels = [("(a) 输入门", I, CMAP_GATE, 0.0, 1.0),
          ("(b) 遗忘门", Fg, CMAP_GATE, 0.0, 1.0),
          ("(c) 输出门", O, CMAP_GATE, 0.0, 1.0),
          ("(d) 内部单元状态", C, CMAP_CELL, None, None)]
fig, axes = plt.subplots(2, 2, figsize=(16, 8))
for ax, (title, M, cmap, vmin, vmax) in zip(axes.flat, panels):
    if vmin is None:                                    # 发散色图对称居中于 0

    im = ax.imshow(M, aspect="auto", cmap=cmap, vmin=vmin, vmax=vmax)
    ax.set_title(title)
    ax.set_xticks(range(len(inputs))); ax.set_xticklabels(inputs)
    ax.set_yticks([])
    ax.set_xlabel("输入数字"); ax.set_ylabel("维度")
    fig.colorbar(im, ax=ax, fraction=0.046, pad=0.04)
fig.tight_layout()
fig.savefig("rnn-gate-vis.pdf", bbox_inches="tight")    # 生成后复制到书稿 chap-循环神经网络/
plt.show()
```

## 5. 梯度截断：让训练更稳

RNN 的 BPTT 会经过 $L$ 步反向传播，梯度容易在某些 step 突然变大。`nn.utils.clip_grad_norm_(params, max_norm)` 会在反传完成后、`opt.step()` 之前，把所有参数梯度的整体 L2 范数缩放到 ≤ `max_norm`。

下面用一个随机初始化（**非 identity**）的 SRN 跑长序列，记录每个 step 的梯度范数。

```python
class WildSRN(nn.Module):
    """Wh 用更大的随机初始化（不是 identity），更容易看到梯度变大的情形。"""
    def __init__(self, embed=32, hidden=64, n_class=19):
        super().__init__()
        self.emb = nn.Embedding(10, embed)
        self.Wx = nn.Parameter(torch.randn(embed,  hidden) / math.sqrt(embed))
        self.Wh = nn.Parameter(torch.randn(hidden, hidden) * 2.0 / math.sqrt(hidden))
        self.b  = nn.Parameter(torch.zeros(hidden))
        self.fc = nn.Linear(hidden, n_class)

    def forward(self, x):
        e = self.emb(x); h = torch.zeros(e.size(0), self.Wh.size(0))
        for t in range(e.size(1)):
            h = torch.tanh(e[:, t] @ self.Wx + h @ self.Wh + self.b)
        return self.fc(h)

@torch.no_grad()
def grad_l2(model):
    s = 0.0
    for p in model.parameters():
        if p.grad is not None:
            s += p.grad.pow(2).sum().item()
    return math.sqrt(s)

def grad_trace(clip, epochs=2, L=50):
    torch.manual_seed(0)
    model = WildSRN()
    opt = optim.SGD(model.parameters(), lr=1.0)
    loss_fn = nn.CrossEntropyLoss()
    X_tr, y_tr = make_digit_sum(1500, L, seed=0)
    loader = DataLoader(TensorDataset(X_tr, y_tr), batch_size=64, shuffle=True)
    trace = []
    for _ in range(epochs):
        for x, y in loader:
            opt.zero_grad(); loss_fn(model(x), y).backward()
            trace.append(grad_l2(model))
            if clip is not None:
                nn.utils.clip_grad_norm_(model.parameters(), clip)
            opt.step()
    return trace

g_no   = grad_trace(clip=None)
g_clip = grad_trace(clip=0.5)

plt.figure(figsize=(9, 4))
plt.plot(g_no,   label='no clip')
plt.plot(g_clip, label='clip=0.5')
plt.axhline(0.5, color='red', lw=0.5, ls='--')
plt.xlabel('step'); plt.ylabel('grad L2 norm before clipping'); plt.yscale('log')
plt.legend(); plt.grid(alpha=0.3); plt.show()
print(f'no clip : max={max(g_no):.3e}   median={sorted(g_no)[len(g_no)//2]:.3e}')
print(f'clip 0.5: max={max(g_clip):.3e}   median={sorted(g_clip)[len(g_clip)//2]:.3e}')
print('(注意：trace 记录的是 clip 之前的值；clip 影响的是 *实际写入参数* 的更新量。)')
```

**说明**：横坐标是 step，纵坐标是**截断前**的梯度 L2 范数（log 轴）。`clip=0.5` 把所有更新步的范数实际写入到优化器之前压到 0.5 以下。`no clip` 偶尔会冒出尖峰甚至更高的值——这就是为什么 RNN/Transformer 训练都默认带 grad clip。

## 小结

- **手写 SRN 与 `nn.RNN` 等价**——但实际项目用 `nn.RNN / nn.LSTM / nn.GRU`，底层 fused 实现快得多。
- **`weight_ih_l0` / `weight_hh_l0`** 的形状是 `[hidden, input/hidden]`，与教科书 / 手写公式互为转置。
- **LSTM > SRN**，但**两者在 L=20+ 都吃力**——长程依赖是 RNN 家族的根本难题。
- **`clip_grad_norm_(params, max_norm)`** 是 RNN/Transformer 训练的标配，几乎不用考虑是否打开。
