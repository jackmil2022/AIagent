### 2.2 知识库索引

```python
from langchain.document_loaders import JSONLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import HuggingFaceEmbeddings
from langchain.vectorstores import Chroma

# 加载文档
loader = JSONLoader(
    data_files="faq.json",
    jq_schema=".[]"
)
documents = loader.load()

# 分块
splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,
    chunk_overlap=50
)
chunks = splitter.split_documents(documents)

# 向量化
embeddings = HuggingFaceEmbeddings(model_name="bge-large-zh-v1.5")

# 存入向量数据库
vectorstore = Chroma.from_documents(
    chunks,
    embeddings,
    persist_directory="./customer_service_db"
)

# 创建检索器
retriever = vectorstore.as_retriever(
    search_type="similarity",
    search_kwargs={"k": 3}
)
```

### 3.2 意图识别实现

```python
from openai import OpenAI

client = OpenAI()

def classify_intent(user_input):
    """识别用户意图"""
    prompt = f"""
你是一个客服意图识别系统。

用户输入：{user_input}

可选意图：
- faq: 常见问题咨询
- complaint: 投诉建议
- order: 订单查询
- refund: 退货退款
- technical: 技术支持
- human: 转人工客服

请只输出意图名称：
"""
    
    response = client.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}],
        temperature=0
    )
    
    intent = response.choices[0].message.content.strip()
    return intent

# 测试
print(classify_intent("我想退货"))  # refund
print(classify_intent("怎么还没到货"))  # order
print(classify_intent("转人工"))  # human
```

```python
    template=prompt_template,
    input_variables=["context", "question"]
)

# 创建问答链
```

## 🔰 5. 多轮对话管理

### 5.1 对话状态

```python
class ConversationState:
    def __init__(self):
        self.user_id = None
        self.intent = None
        self.context = []
        self.order_id = None
        self.stage = "greeting"
    
    def add_message(self, role, content):
        self.context.append({
            "role": role,
            "content": content
        })
    
    def get_context(self, max_messages=10):
        return self.context[-max_messages:]
```

### 5.2 对话管理器

```python
class CustomerServiceBot:
    def __init__(self):
        self.conversations = {}
        self.qa_chain = qa_chain
        self.intent_classifier = classify_intent
    
    def get_or_create_conversation(self, user_id):
        if user_id not in self.conversations:
            self.conversations[user_id] = ConversationState()
            self.conversations[user_id].user_id = user_id
        return self.conversations[user_id]
    
    def chat(self, user_id, user_input):
        """处理用户输入"""
        state = self.get_or_create_conversation(user_id)
        
        # 添加用户消息
        state.add_message("user", user_input)
        
        # 意图识别
        intent = self.intent_classifier(user_input)
        state.intent = intent
        
        # 根据意图分发处理
        if intent == "human":
            return self._transfer_to_human(state)
        elif intent == "faq":
            return self._handle_faq(state, user_input)
        elif intent == "refund":
            return self._handle_refund(state, user_input)
        elif intent == "order":
            return self._handle_order(state, user_input)
        else:
            return self._handle_general(state, user_input)
    
    def _handle_faq(self, state, user_input):
        """处理常见问题"""
        result = self.qa_chain({"query": user_input})
        answer = result["result"]
        state.add_message("assistant", answer)
        return answer
    
    def _transfer_to_human(self, state):
        """转人工"""
        return "好的，正在为您转接人工客服，请稍等..."
    
    def _handle_refund(self, state, user_input):
        """处理退货"""
        # 查询订单
        # 创建工单
        return "我理解您需要退货，请提供订单号，我来帮您处理。"
    
    def _handle_order(self, state, user_input):
        """处理订单查询"""
        return "请提供您的订单号，我帮您查询物流信息。"
```

## 🔰 6. 工单系统

### 6.1 工单模型

```python
from dataclasses import dataclass
from datetime import datetime
from typing import Optional

@dataclass
class Ticket:
    id: str
    user_id: str
    intent: str
    description: str
    status: str = "open"
    created_at: datetime = None
    assigned_to: Optional[str] = None
    
    def __post_init__(self):
        if self.created_at is None:
            self.created_at = datetime.now()

class TicketManager:
    def __init__(self):
        self.tickets = {}
    
    def create_ticket(self, user_id, intent, description):
        ticket_id = f"TK{len(self.tickets) + 1:06d}"
        ticket = Ticket(
            id=ticket_id,
            user_id=user_id,
            intent=intent,
            description=description
        )
        self.tickets[ticket_id] = ticket
        return ticket
    
    def get_ticket(self, ticket_id):
        return self.tickets.get(ticket_id)
    
    def update_status(self, ticket_id, status):
        if ticket_id in self.tickets:
            self.tickets[ticket_id].status = status
```
