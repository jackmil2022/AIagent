# 多轮对话

# 05.多轮对话.py

```python
# 从 openai 库中导入 OpenAI 类，用于调用 DeepSeek 的 Chat API
from openai import OpenAI  
# 导入 os 模块，用于访问系统环境变量
import os  

# 从系统环境变量中获取名为 DEEPSEEK_API_KEY 的 API 密钥
api_key = os.environ['DEEPSEEK_API_KEY']  

# 创建一个 OpenAI 客户端对象，指定使用 DeepSeek 的 API 网关地址，并传入 API 密钥进行鉴权
client = OpenAI(base_url='https://api.deepseek.com', api_key=api_key)

# 构建初始对话消息，表示用户向模型提出第一个问题
messages = [{'role': 'user', 'content': '1+1等于几'}]

# 向 DeepSeek 模型发送聊天请求，传入模型名和消息历史
response = client.chat.completions.create(
    model='deepseek-chat',  # 指定要使用的 DeepSeek 聊天模型
    messages=messages       # 发送的消息列表（对话上下文）
)

# 将模型的回复（AI 回复）加入到消息列表中，便于维持对话上下文
messages.append(response.choices[0].message)

# 打印模型的第一轮回复内容
print(f'Messages Round 1: {response.choices[0].message}')

# 用户继续提问“再加1呢”，将此问题加入消息列表，维持上下文
messages.append({'role': 'user', 'content': '再加1呢'})

# 再次向模型发送完整的对话历史，以获取新一轮回复
response = client.chat.completions.create(
    model='deepseek-chat',  # 使用相同的 DeepSeek 模型
    messages=messages        # 发送包含两轮对话的消息列表
)

# 将模型的第二次回复加入消息列表
messages.append(response.choices[0].message)

# 打印模型的第二轮回复内容
print(f'Messages Round 2:{response.choices[0].message}')
```
